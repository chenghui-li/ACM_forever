vector<int> printListFromTailToHead(ListNode* head) {
        vector<int> ans;
        if(head == nullptr)
            return ans;
        stack<int> s;
        ListNode *now = head;
        while(now != nullptr){
            s.push(now->val);
            now = now->next;
        }
        while(!s.empty()){
            ans.push_back(s.top());
            s.pop();
        }
        return ans;
    }

void getnow(vector<int> *ans,ListNode *head){
        if(head->next != nullptr)
            getnow(ans,head->next);
